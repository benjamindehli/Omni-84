# omni-84-plugin — the distributable product (VST3 / AU / Standalone).
# Thin wrapper around dehli_musikk_sampler_engine (dm::ManifestPluginProcessor).

juce_add_plugin(Omni84
    COMPANY_NAME            "DehliMusikk"
    BUNDLE_ID               com.dehlimusikk.omni84
    VERSION                 0.1.0
    PLUGIN_MANUFACTURER_CODE Dhmk
    PLUGIN_CODE              Om84
    FORMATS                 VST3 AU Standalone
    PRODUCT_NAME            "Omni-84"
    IS_SYNTH                TRUE
    NEEDS_MIDI_INPUT        TRUE
    NEEDS_MIDI_OUTPUT       FALSE
    IS_MIDI_EFFECT          FALSE
    EDITOR_WANTS_KEYBOARD_FOCUS FALSE
    COPY_PLUGIN_AFTER_BUILD TRUE
    VST3_CATEGORIES         "Instrument" "Sampler"
    AU_MAIN_TYPE            "kAudioUnitType_MusicDevice"
)

target_sources(Omni84 PRIVATE
    source/Plugin.cpp
)

target_compile_definitions(Omni84 PUBLIC
    JUCE_WEB_BROWSER=0
    JUCE_USE_CURL=0
    JUCE_VST3_CAN_REPLACE_VST2=0
)

target_link_libraries(Omni84 PRIVATE
    dehli_musikk_sampler_engine
    juce::juce_audio_utils
    juce::juce_recommended_config_flags
    juce::juce_recommended_warning_flags
)

# Embedded asset bundle: the engine's JSON manifest + FLAC samples, generated by
# ds-plugin-converter into assets/ (run `dmse_convert <library> omni-84-plugin/assets`).
# Embedded via juce_add_binary_data; the processor reads them from BinaryData at load.
# The manifest is embedded SPLIT (manifest/index.json + modes/ + optional partials/)
# and merged at load — NOT the single manifest.json. Mode slugs and partial names
# share one flat resource namespace, so they must not collide (they don't for Omni-84).
file(GLOB OMNI84_ASSETS CONFIGURE_DEPENDS
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/manifest/index.json"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/manifest/modes/*.json"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/manifest/partials/*.json"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/samples/*.flac"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/ir/*.flac"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/images/*.png"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/images/*.jpg"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/images/*.jpeg")

if(OMNI84_ASSETS)
    juce_add_binary_data(Omni84Assets SOURCES ${OMNI84_ASSETS})
    target_link_libraries(Omni84 PRIVATE Omni84Assets)
    target_compile_definitions(Omni84 PRIVATE OMNI84_HAS_ASSETS=1)
    list(LENGTH OMNI84_ASSETS OMNI84_ASSET_COUNT)
    message(STATUS "Omni84: embedding ${OMNI84_ASSET_COUNT} asset file(s) from assets/")
else()
    message(WARNING "Omni84: no assets in assets/ — plugin will be silent. "
                    "Generate them with the dmse_convert target first.")
endif()

# Install the disk sample pack (default) into the shared support folder the engine reads.
dmse_setup_plugin(Omni84 "Omni-84")
